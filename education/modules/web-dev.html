Express.js skeleton

import express from "express";
const app = express();
app.use(express.json());

app.get("/api/health", (req, res) => res.json({ok:true}));

app.post("/api/items", (req, res, next) => {
  try {
    const {name} = req.body;
    if (!name) return res.status(400).json({error:"name required"});
    // save...
    res.status(201).json({id: 1, name});
  } catch (e) { next(e); }
});

app.use((err, req, res, _next) => res.status(500).json({error:"server"}));
app.listen(8080);

Flask micro-API

from flask import Flask, request, jsonify
app = Flask(__name__)

@app.get("/api/health")
def health(): return jsonify(ok=True)

@app.post("/api/items")
def create_item():
  data = request.get_json() or {}
  if "name" not in data: return jsonify(error="name required"), 400
  return jsonify(id=1, name=data["name"]), 201

if __name__ == "__main__":
  app.run(port=8080)

Security checklist

  • Hash passwords (bcrypt/argon2); never store plain text.
  • Validate + sanitise inputs; encode outputs.
  • CSRF tokens for state-changing form posts.
  • Use parameterised SQL / ORM to prevent injection.
  • Set secure cookies (HttpOnly, Secure, SameSite).
