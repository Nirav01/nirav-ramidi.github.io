Quick Reference

Pointers & allocation

#include <stdlib.h>
int *a = malloc(10 * sizeof *a);
if (!a) { /* handle OOM */ }
a[0] = 42;
/* ... */
free(a); a = NULL;

Always check allocations; null out freed pointers.

Struct + API split

// point.h
#ifndef POINT_H
#define POINT_H
typedef struct { double x, y; } Point;
double dist(Point a, Point b);
#endif

// point.c
#include "point.h"
#include <math.h>
double dist(Point a, Point b){
  double dx=a.x-b.x, dy=a.y-b.y;
  return sqrt(dx*dx + dy*dy);
}

Header guards prevent duplicate definitions during compilation.

File I/O (binary)

#include <stdio.h>
FILE *f = fopen("data.bin", "rb");
if (!f) return 1;
size_t n = fread(buf, 1, sizeof buf, f);
fclose(f);

Check return values of I/O calls; handle short reads/writes.

Make + GDB

# Makefile (mini)
CFLAGS=-Wall -Wextra -O2 -g
all: app
app: main.o list.o
	$(CC) $(CFLAGS) -o $@ $^
clean: ; rm -f *.o app

# Debug
gdb ./app
(gdb) break main
(gdb) run
(gdb) bt  # backtrace

Compile with -g for symbols; use valgrind --leak-check=full ./app to find leaks.
